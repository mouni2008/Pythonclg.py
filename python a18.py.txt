
class Solution:
    def addTwoNumbers(self, l1, l2):
        dummy = ListNode(0)
        temp = dummy
        carry = 0

        while l1 != None or l2 != None or carry != 0:

            sum = carry

            if l1 != None:
                sum = sum + l1.val
                l1 = l1.next

            if l2 != None:
                sum = sum + l2.val
                l2 = l2.next

            carry = sum // 10
            newNode = ListNode(sum % 10)
            temp.next = newNode
            temp = temp.next

        return dummy.next
